\begin{flushleft}
\begin{longtable}[l]{ll} %% [l] tabulka je zarovnana vlevo; [c] zarovnani na stred; [r] zarovnani v pravo
  $\Rbb$                 & množina všech reálných čísel \\[1mm]
  $\Zbb$                 & množina všech celých čísel \\[1mm]
  $\mathbb{P}$           & množina všech prvočísel\\[1mm]
\end{longtable}
\end{flushleft}
